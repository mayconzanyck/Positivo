ALGORITMO repeticao
VAR
    INTEIRO: contador
INICIO
    contador <- 1;

    ENQUANTO(contador <= 5) FAÇA
        ESCREVA(contador);
        contador <- contador + 1;
    FIMENQUANTO
FIM
